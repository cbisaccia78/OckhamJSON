bool parse(nAryTreeNode root){
    if(nodeIsLeaf(root)){
        parseKeyOffSet(root);
        if(root.isValid){
            parseKey(root);
            if(root.isValid){
                parseValueOffset(root);
                if(root.isValid){
                    parseValue(root);
                    return root.isValid;
                }
            }
        }
        return false;
    }
    bool validChildren = true;
    for(child in root.children){
        validChildren = validChildren && parse(child)
    }
    return validChildren;
}
